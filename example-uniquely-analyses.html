3.2.3 All network in our database

gene_metabolite <- gene_metabolite %>%
  dplyr::select(-"pathway_type") %>%
  unique()

nodes1 <- gene_metabolite[,1:2]
nodes2 <- gene_metabolite[,3:4]
names(nodes1) <- c("type","name")
names(nodes2) <- c("type","name")
nodes <- rbind(nodes1,nodes2) %>%
  unique() %>%
  dplyr::select(c("name","type"))

network <- igraph::graph_from_data_frame(d=gene_metabolite[,c(2,4)], vertices=nodes,directed=F)
#plot(network)

name <- data.frame(name=igraph::V(network)$name) %>%
  dplyr::left_join(nodes,by="name")
coul  <- c("blue","red")
my_color <- coul[as.numeric(as.factor(name$type))]
#my_color

shape1=c("square","circle")
my_shape <- shape1[as.numeric(as.factor(name$type))]

#计算每个节点的数量
deg <- igraph::degree(network, mode="all")

plot(network,vertex.color=my_color,vertex.shape=my_shape,vertex.size=deg**0.3,
     vertex.label.cex=0.2)
